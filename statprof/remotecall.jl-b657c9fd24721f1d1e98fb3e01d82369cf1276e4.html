69 function test_existing_ref(r::AbstractRemoteRef)
70 found = getkey(client_refs, r, nothing)
71 if found !== nothing
72 @assert r.where > 0
73 if isa(r, Future)
74 # this is only for copying the reference from Future to RemoteRef (just created)
75 fv_cache = @atomic :acquire found.v
76 rv_cache = @atomic :monotonic r.v
77 if fv_cache === nothing && rv_cache !== nothing
78 # we have recd the value from another source, probably a deserialized ref, send a del_client message
79 send_del_client(r)
80 @lock found.lock begin
81 @atomicreplace found.v nothing => rv_cache
82 end
83 end
84 end
85 return found::typeof(r)
86 end
87
88 client_refs[r] = nothing
89 finalizer(finalize_ref, r)
90 return r
91 end

93 function finalize_ref(r::AbstractRemoteRef)
94 if r.where > 0 # Handle the case of the finalizer having been called manually
95 if trylock(client_refs.lock) # trylock doesn't call wait which causes yields
96 try
97 delete!(client_refs.ht, r) # direct removal avoiding locks
98 if isa(r, RemoteChannel)
99 send_del_client_no_lock(r)
100 else
101 # send_del_client only if the reference has not been set
102 v_cache = @atomic :monotonic r.v
103 v_cache === nothing && send_del_client_no_lock(r)
104 @atomic :monotonic r.v = nothing
105 end
106 r.where = 0
107 finally
108 unlock(client_refs.lock)
109 end
110 else
111 finalizer(finalize_ref, r)
112 return nothing
113 end
114 end

142 """
143 remoteref_id(r::AbstractRemoteRef) -> RRID
144
145 `Future`s and `RemoteChannel`s are identified by fields:
146
147 * `where` - refers to the node where the underlying object/storage
148 referred to by the reference actually exists.
149
150 * `whence` - refers to the node the remote reference was created from.
151 Note that this is different from the node where the underlying object
152 referred to actually exists. For example calling `RemoteChannel(2)`
153 from the master process would result in a `where` value of 2 and
154 a `whence` value of 1.
155
156 * `id` is unique across all references created from the worker specified by `whence`.
157
158 Taken together, `whence` and `id` uniquely identify a reference across all workers.
159
160 `remoteref_id` is a low-level API which returns a `RRID`
161 object that wraps `whence` and `id` values of a remote reference.
162 """
163 remoteref_id(r::AbstractRemoteRef) = RRID(r.whence, r.id)

595 """
596 fetch(x::Future)
597
598 Wait for and get the value of a [`Future`](@ref). The fetched value is cached locally.
599 Further calls to `fetch` on the same reference return the cached value. If the remote value
600 is an exception, throws a [`RemoteException`](@ref) which captures the remote exception and backtrace.
601 """
602 function fetch(r::Future)
603 v_cache = @atomic r.v
604 v_cache !== nothing && return something(v_cache)
605
606 if r.where == myid()
607 rv, v_cache = @lock r.lock begin
608 v_cache = @atomic :monotonic r.v
609 rv = v_cache === nothing ? lookup_ref(remoteref_id(r)) : nothing
610 rv, v_cache
611 end
612
613 if v_cache !== nothing
614 return something(v_cache)
615 else
616 v_local = fetch(rv.c)
617 end
618 else
619 v_local = call_on_owner(fetch_ref, r)
620 end
621
622 v_cache = @atomic r.v
623
624 if v_cache === nothing # call_on_owner case
625 v_old, status = @lock r.lock begin
626 @atomicreplace r.v nothing => Some(v_local)
627 end
628 # status == true - when value obtained through call_on_owner
629 # status == false - any other situation: atomicreplace fails, because by the time the lock is obtained cache will be populated
630 # why? local put! performs caching and putting into channel under r.lock
631
632 # for local put! use the cached value, for call_on_owner cases just take the v_local as it was just cached in r.v
633
634 # remote calls getting the value from `call_on_owner` used to return the value directly without wrapping it in `Some(x)`
635 # so we're doing the same thing here
636 if status
637 send_del_client(r)
638 return v_local
639 else # this `v_cache` is returned at the end of the function
640 v_cache = v_old
641 end
642 end
643
644 send_del_client(r)
645 something(v_cache)
646 end

660 """
661 put!(rr::Future, v)
662
663 Store a value to a [`Future`](@ref) `rr`.
664 `Future`s are write-once remote references.
665 A `put!` on an already set `Future` throws an `Exception`.
666 All asynchronous remote calls return `Future`s and set the

669 function put!(r::Future, v)
670 if r.where == myid()
671 rid = remoteref_id(r)
672 rv = lookup_ref(rid)
673 isready(rv) && error("Future can be set only once")
674 @lock r.lock begin
675 put!(rv, v) # this notifies the tasks waiting on the channel in fetch
676 set_future_cache(r, v) # set the cache before leaving the lock, so that the notified tasks already see it cached
677 end
678 del_client(rid, myid())
679 else
680 @lock r.lock begin # same idea as above if there were any local tasks fetching on this Future
681 call_on_owner(put_future, r, v, myid())
682 set_future_cache(r, v)
683 end
684 end
685 r

725
726 take!(rv::RemoteValue, args...) = take!(rv.c, args...)
727 function take_ref(rid, caller, args...)
728 rv = lookup_ref(rid)
729 synctake = false
730 if myid() != caller && rv.synctake !== nothing
731 # special handling for local put! / remote take! on unbuffered channel
732 # github issue #29932
733 synctake = true
734 lock(rv.synctake)
735 end
736
737 v = try
738 take!(rv, args...)
739 catch e
740 # avoid unmatched unlock when exception occurs
741 # github issue #33972
742 synctake && unlock(rv.synctake)
743 rethrow(e)
744 end
745
746 isa(v, RemoteException) && (myid() == caller) && throw(v)
747
748 if synctake
749 return SyncTake(v, rv)
750 else
751 return v
752 end
753 end
