47 const SLOPE = 0.005
48 const WAVE_START = 300.0 # nm
49 const WAVE_END = 1000.0 # nm
50 const CENTRAL = 555.6 # nm
51 const VAL = SLOPE / (WAVE_END - WAVE_START)
52 const SUPERCAL_UNCERTAINTY = 1.0 / 3.0

54 function CovarianceMatrix(instruments::Dict{String,Instrument})
55 filters = Dict{String,Filter}()
56 for instrument in values(instruments)
57 merge!(filters, instrument.filters)
58 end
59 ndim = length(filters)
60 keys = Vector{String}(undef, ndim)
61 ZPUncertainty = zeros(Float64, ndim, ndim)
62 FilterUncertainty = Vector{Float64}(undef, ndim)
63 for (i, (filter_name, filter_i)) in enumerate(filters)
64 keys[i] = filter_name
65 SuperCalCorrection = (((1 + SUPERCAL_UNCERTAINTY) * filter_i.SuperCalCorrection)^2)
66 # Filter Uncertainty already squared
67 FilterUncertainty[i] = filter_i.FilterUncertainty + SuperCalCorrection
68 for (j, (filter_name_j, filter_j)) in enumerate(filters)
69 if i == j
70 # Already squared
71 ZPUncertainty[i, j] = filter_i.ZPUncertainty
72 end
73 diff_i = filter_i.CentralWavelength - CENTRAL
74 diff_j = filter_j.CentralWavelength - CENTRAL
75 ZPUncertainty[i, j] += VAL * VAL * diff_i * diff_j
76 end
77 end
78 return CovarianceMatrix(keys, ZPUncertainty, FilterUncertainty)
79 end

91 function generateMatrix(covariance::CovarianceMatrix)
92 nfilt = length(covariance.FilterUncertainty)
93 FilterUncertaintyMatrix = diagm(covariance.FilterUncertainty)
94 ZeroMatrix = zeros(Float64, nfilt, nfilt)
95 cov_matrix = [FilterUncertaintyMatrix ZeroMatrix; ZeroMatrix covariance.ZPUncertainty]
96 if !(isequal(cov_matrix, transpose(cov_matrix)))
97 if isapprox(cov_matrix, transpose(cov_matrix))
98 @info "Matrix is not symmetric, but is approximately symmetric, fixing."
99 cov_matrix = 0.5 * (cov_matrix + transpose(cov_matrix))
100 else
101 @info "Matrix is not symmetric"
102 non_symmetric_indices = findall(cov_matrix .!= transpose(cov_matrix))
103 for (i, j) in Tuple.(non_symmetric_indices)
104 @debug "i: $i, j: j"
105 @debug "Covariance[i,j]: $(cov_matrix[i, j])"
106 @debug "Covariance[j,i]: $(cov_matrix[j, i])"
107 end
108 end
109 end
110 eigen_values = eigvals(cov_matrix)
111 if any(eigen_values .< 0)
112 @info "Matrix is not positive definite, fixing."
113 offset_matrix = I * (eps(Float64) - minimum(eigen_values))
114 @debug "$offset_matrix"
115 cov_matrix += offset_matrix
116 end
117 return cov_matrix
118 end
