282 # We sometimes need to compute checksums. We do this by first calling begin_checksum when
283 # starting to handle whatever needs checksumming, and calling end_checksum afterwards. Note
284 # that we never compute nested checksums, but we may compute multiple checksums
285 # simultaneously. This strategy is not thread-safe.
286
287 function begin_checksum_read(io::MmapIO)
288 idx = io.nchecksum += 1
289 if idx > length(io.checksum_pos)
290 push!(io.checksum_pos, position(io))
291 else
292 @inbounds io.checksum_pos[idx] = position(io)
293 end
294 io
295 end
296 function begin_checksum_write(io::MmapIO, sz::Integer)
297 ensureroom(io, sz)
298 begin_checksum_read(io)
299 end
300 function end_checksum(io::MmapIO)
301 @inbounds v = io.checksum_pos[io.nchecksum]
302 io.nchecksum -= 1
303 Lookup3.hash(Ptr{UInt8}(io.startptr + v), position(io) - v)
304 end
