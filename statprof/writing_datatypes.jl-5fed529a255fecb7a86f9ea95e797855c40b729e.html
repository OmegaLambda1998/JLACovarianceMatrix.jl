52 # Determines whether a type will have the same layout on disk as in memory
53 function samelayout(T::DataType)
54 isempty(T.types) && return true
55 offset = 0
56 for i = 1:length(T.types)
57 offset != fieldoffset(T, i) && return false
58 ty = T.types[i]
59 ty !== writeas(ty) && return false
60 !samelayout(ty) && return false
61 offset += sizeof(ty)
62 end
63 return offset == sizeof(T)
64 end

560 # jlconvert for empty objects
561 function jlconvert(::ReadRepresentation{T,nothing}, f::JLDFile, ptr::Ptr,
562 header_offset::RelOffset) where T
563 sizeof(T) == 0 && return newstruct(T)::T
564
565 # In this case, T is a non-empty object, but the written data was empty
566 # because the custom serializers for the fields all resulted in empty
567 # objects
568 fields = map(T.types) do ty
569 writtenas = writeas(ty)
570 @assert sizeof(writtenas) == 0
571 if writtenas === ty
572 # This will usually equal `ty()` unless ty does not have a
573 # constructor without arguments
574 jlconvert(ReadRepresentation{ty,nothing}(), f, ptr, header_offset)
575 else
576 rconvert(ty,
577 jlconvert(ReadRepresentation{writtenas,nothing}(), f, ptr, header_offset)
578 )
579 end
580 end
581 if T <: Tuple
582 # Tuples are weird in that you can't instantiate them with Tuple{T,S}(t,s)

584 end

588 # odr gives the on-disk representation of a given type, similar to
589 # fieldodr, but actually encoding the data for things that odr stores
590 # as references
591 function odr(::Type{T}) where T
592 if !hasdata(T)
593 # A pointer singleton or ghost. We need to write something, but we'll
594 # just write a single byte.
595 return nothing
596 elseif isbitstype(T) && samelayout(T)
597 # Has a specialized convert method or is an unpadded type
598 return T
599 end
600
601 offsets = zeros(Int, length(T.types))
602 odrs = Vector{Any}(undef, length(T.types))
603 offset = 0
604 for i = 1:length(T.types)
605 ty = T.types[i]
606 writtenas = writeas(ty)
607 fodr = fieldodr(writtenas, i <= ninitialized(T))
608 if writtenas !== ty && fodr !== nothing
609 odrs[i] = CustomSerialization{writtenas,fodr}
610 else
611 odrs[i] = fodr
612 end
613 offsets[i] = offset
614 offset += odr_sizeof(fodr)
615 end
616
617 OnDiskRepresentation{(offsets...,), Tuple{T.types...}, Tuple{odrs...}, offset}()
618 end
