10 function load_dataset(f::JLDFile, offset::RelOffset)
11 if haskey(f.jloffset, offset)
12 # Stored as WeakRefs and may no longer exist
13 val = f.jloffset[offset].value
14 val !== nothing && return val
15 end
16 if isgroup(f, offset)
17 return let loaded_groups = f.loaded_groups
18 get!(()->load_group(f, offset), loaded_groups, offset)

23 chunk_start::Int64 = fileoffset(f, offset)
24 seek(io, chunk_start)
25 # Version 1 object header have no signature and start with version
26 header_version = jlread(io, UInt8)

29 if header_version == 1

31 cio = io
32 sz,_,groupflags = read_obj_start(cio)
33 chunk_end = position(cio) + sz
34 # Skip to nearest 8byte aligned position
35 skip_to_aligned!(cio, fileoffset(f, offset))
36 else
37 header_version = 2

667
668 function Base.delete!(g::Group, name::AbstractString)
669 g.f.writable || throw(ArgumentError("Cannot delete in read-only mode"))
670 if !haskey(g, name)
671 @warn "No entry named $name was found"
672 return
673 end
674
675 if '/' in name
676 dir, dname = rsplit(name, '/'; limit=2)
677 if isempty(dir)
678 g = g.f.root_group
679 else
680 g = g[dir]
681 end
682 name = string(dname)
683 end
684
685 # Simple case first. If it hasn't been written yet,
686 # the file doesn't need to be altered.
687 if haskey(g.unwritten_links, name)
688 delete!(g.unwritten_links, name)
689 return
690 elseif haskey(g.unwritten_child_groups, name)
691 delete!(g.unwritten_child_groups, name)
692 return
693 end
694
695 # Dataset must already exist in the file
696 # Retrieve offset of group in file
697 offset = group_offset(g)
698 offset == NULL_REFERENCE && throw(InternalError("Group could not be found."))
699 delete_written_link!(g.f, offset, name)
700 delete!(g.written_links, name)
701 return
702 end

704
705 function group_offset(g::Group)
706 # Given a group, retrieve its offset inside the parent file
707 # Return a null reference when it can't be found. However, that
708 # should never be the case as the calling functions checks for this
709 if g === g.f.root_group
710 return g.f.root_group_offset
711 end
712 for (offset, lg) in pairs(g.f.loaded_groups)
713 if lg === g
714 return offset
715 end
716 end
717 return NULL_REFERENCE
718 end
719
720 function delete_written_link!(f::JLDFile, roffset::RelOffset, name::AbstractString)
721 # Location of written link in group structure is not known a priory
722 # Instead, we walk through the structure similar to `load_group`
723 # until the correct link message is found
724 # The deletion is done by replacing that message by a placeholder nill message
725 io = f.io
726 chunk_start_offset::Int64 = fileoffset(f, roffset)
727 seek(io, chunk_start_offset)
728
729 sz, = read_obj_start(io)
730 chunk_checksum_offset::Int64 = position(io) + sz
731
732 continuation_offset::Int64 = -1
733 continuation_length::Length = 0
734 link_deleted = false
735 while !link_deleted
736 if continuation_offset != -1
737 seek(io, continuation_offset)
738 chunk_start_offset = continuation_offset
739 chunk_checksum_offset = continuation_offset + continuation_length - 4
740 continuation_offset = -1
741 jlread(io, UInt32) == OBJECT_HEADER_CONTINUATION_SIGNATURE || throw(InvalidDataException())
742 end
743
744 while (curpos = position(io)) <= chunk_checksum_offset - 4
745 msg = jlread(io, HeaderMessage)
746 endpos = curpos + jlsizeof(HeaderMessage) + msg.size
747
748 if msg.msg_type == HM_LINK_MESSAGE
749 dataset_name, loffset = read_link(io)
750 if dataset_name == name
751 # delete link
752 seek(io, curpos)
753 jlwrite(io, HeaderMessage(HM_NIL, msg.size, 0))
754 link_deleted = true
755 break
756 end
757 elseif msg.msg_type == HM_OBJECT_HEADER_CONTINUATION
758 continuation_offset = chunk_start_offset = fileoffset(f, jlread(io, RelOffset))
759 continuation_length = jlread(io, Length)
760 end
761 seek(io, endpos)
762 end
763
764 continuation_offset == -1 && break
765 end
766
767 # Update the Checksum
768 seek(io, chunk_start_offset)
769 cio = begin_checksum_read(io)
770 seek(cio, chunk_checksum_offset)
771 seek(io, chunk_checksum_offset)
772 jlwrite(io, end_checksum(cio))
773 return
774 end
