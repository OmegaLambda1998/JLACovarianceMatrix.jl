43 rr::ReadRepresentation{T,T}) where T
44 io = f.io
45 inptr = io.curptr
46 n = length(v)
47 nb = odr_sizeof(T)*n
48 if nb > MMAP_CUTOFF && (!Sys.iswindows() || !f.written)
49 # It turns out that regular IO is faster here (at least on OS X), but on Windows,
50 # we shouldn't use ordinary IO to read, since coherency with the memory map is not
51 # guaranteed
52 mmapio = f.io

102 function raw_write(io::MmapIO, ptr::Ptr{UInt8}, nb::Int)
103 if nb > MMAP_CUTOFF
104 pos = position(io)
105
106 # Ensure that the current page has been flushed to disk
107 msync(io, pos, min(io.endptr - io.curptr, nb))
108
109 # Write to the underlying IOStream
110 regulario = io.f
111 seek(regulario, pos)
112 unsafe_write(regulario, ptr, nb)
113
114 # Invalidate cache of any pages that were just written to
115 msync(io, pos, min(io.n - pos, nb), true)
116
117 # Make sure the mapping is encompasses the written data
118 ensureroom(io, nb + 1)
119
120 # Seek to the place we just wrote
121 seek(io, pos + nb)
122 else
123 unsafe_write(io, ptr, nb)
124 end
125 nothing
126 end
127 else
128 # Don't use ordinary IO to write files on Windows, since coherency with memory map is
129 # not guaranteed
130 function raw_write(io::MmapIO, ptr::Ptr{UInt8}, nb::Int)
131 unsafe_write(io, ptr, nb)
132 nothing
133 end
134 end
